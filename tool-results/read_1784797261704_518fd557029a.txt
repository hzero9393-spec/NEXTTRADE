   125→// ─── Priority & Status configs ───────────────────────────────────────────────
   126→const priorityConfig: Record<TicketPriority, { color: string; label: string }> = {
   127→  HIGH: { color: '#d44a2d', label: 'High' },
   128→  MEDIUM: { color: '#F59E0B', label: 'Medium' },
   129→  LOW: { color: '#3B82F6', label: 'Low' },
   130→}
   131→
   132→const statusConfig: Record<TicketStatus, { color: string; bg: string; border: string; label: string; icon: React.ComponentType<{ className?: string }> }> = {
   133→  OPEN: { color: '#3B82F6', bg: '#3B82F6/10', border: '#3B82F6/30', label: 'Open', icon: CircleDot },
   134→  IN_PROGRESS: { color: '#F59E0B', bg: '#F59E0B/10', border: '#F59E0B/30', label: 'In Progress', icon: Clock },
   135→  RESOLVED: { color: '#00a87d', bg: '#00a87d/10', border: '#00a87d/30', label: 'Resolved', icon: CheckCircle },
   136→  CLOSED: { color: 'var(--muted-foreground)', bg: '#6b7280/10', border: '#6b7280/30', label: 'Closed', icon: XCircle },
   137→}

   139→// ─── Tickets Page Inner ──────────────────────────────────────────────────────
   140→function TicketsPageInner() {
   141→  const [tickets, setTickets] = useState<SupportTicket[]>([])
   142→  const [loading, setLoading] = useState(true)
   143→  const [search, setSearch] = useState('')
   144→  const [statusFilter, setStatusFilter] = useState<string>('All')
   145→  const [page, setPage] = useState(1)
   146→  const [selectedTicket, setSelectedTicket] = useState<SupportTicket | null>(null)
   147→  const [replyText, setReplyText] = useState('')
   148→  const [sendingReply, setSendingReply] = useState(false)
   149→  const [changingStatus, setChangingStatus] = useState(false)
   150→  const limit = 10
   151→
   152→  const fetchTickets = useCallback(async () => {
   153→    setLoading(true)
   154→    try {
   155→      const res = await adminApi('/tickets')
   156→      const data = await res.json()
   157→      const mapped = (data.tickets || []).map((t: any) => ({
   158→        id: t.id,
   159→        userId: t.userId,
   160→        userName: t.userName || t.user?.name || 'Unknown',
   161→        userEmail: t.userEmail || t.user?.email || '',
   162→        subject: t.subject,
   163→        description: t.description,
   164→        priority: t.priority,
   165→        status: t.status,
   166→        createdAt: t.createdAt,
   167→        updatedAt: t.updatedAt,
   168→        replies: (t.replies || []).map((r: any) => ({
   169→          id: r.id,
   170→          message: r.message,
   171→          isAdmin: r.isAdmin,
   172→          createdAt: r.createdAt,
   173→        })),
   174→      }))
   175→      setTickets(mapped)
   176→    } catch {
   177→      console.warn('[TicketsPage] API fetch failed, using mock data')
   178→      setTickets(mockTickets)
   179→    } finally {
   180→      setLoading(false)
   181→    }
   182→  }, [])
   183→
   184→  useEffect(() => { fetchTickets() }, [fetchTickets])
   185→
   186→  // Filter
   187→  const filteredTickets = tickets.filter((t) => {
   188→    if (statusFilter !== 'All' && t.status !== statusFilter) return false
   189→    if (search) {
   190→      const q = search.toLowerCase()
   191→      return t.subject.toLowerCase().includes(q) || t.userName.toLowerCase().includes(q) || t.id.toLowerCase().includes(q)
   192→    }
   193→    return true
   194→  })
   195→
   196→  // Pagination
   197→  const totalPages = Math.ceil(filteredTickets.length / limit)
   198→  const paginatedTickets = filteredTickets.slice((page - 1) * limit, page * limit)
   199→
   200→  // Stats
   201→  const openCount = tickets.filter(t => t.status === 'OPEN').length
   202→  const inProgressCount = tickets.filter(t => t.status === 'IN_PROGRESS').length
   203→  const resolvedCount = tickets.filter(t => t.status === 'RESOLVED').length
   204→
   205→  const handleSendReply = async () => {
   206→    if (!replyText.trim() || !selectedTicket) return
   207→    setSendingReply(true)
   208→    try {
   209→      await adminApi(`/tickets/${selectedTicket.id}`, {
   210→        method: 'POST',
   211→        body: JSON.stringify({ message: replyText, isAdmin: true }),
   212→      })
   213→      const newReply = { id: `r_${Date.now()}`, message: replyText, isAdmin: true, createdAt: new Date().toISOString() }
   214→      const updated = { ...selectedTicket, replies: [...selectedTicket.replies, newReply] }
   215→      setSelectedTicket(updated)
   216→      setTickets(prev => prev.map(t => t.id === updated.id ? updated : t))
   217→      setReplyText('')
   218→      toast.success('Reply sent')
   219→    } catch {
   220→      const newReply = { id: `r_${Date.now()}`, message: replyText, isAdmin: true, createdAt: new Date().toISOString() }
   221→      const updated = { ...selectedTicket, replies: [...selectedTicket.replies, newReply] }
   222→      setSelectedTicket(updated)
   223→      setTickets(prev => prev.map(t => t.id === updated.id ? updated : t))
   224→      setReplyText('')
   225→      toast.success('Reply sent')
   226→    } finally {
   227→      setSendingReply(false)
   228→    }
   229→  }
   230→
   231→  const handleStatusChange = async (newStatus: string) => {
   232→    if (!selectedTicket) return
   233→    setChangingStatus(true)
   234→    try {
   235→      await adminApi(`/tickets/${selectedTicket.id}`, {
   236→        method: 'PUT',
   237→        body: JSON.stringify({ status: newStatus }),
   238→      })
   239→      const updated = { ...selectedTicket, status: newStatus as TicketStatus, updatedAt: new Date().toISOString() }
   240→      setSelectedTicket(updated)
   241→      setTickets(prev => prev.map(t => t.id === updated.id ? updated : t))
   242→      toast.success(`Ticket status changed to ${statusConfig[newStatus as TicketStatus]?.label || newStatus}`)
   243→    } catch {
   244→      const updated = { ...selectedTicket, status: newStatus as TicketStatus, updatedAt: new Date().toISOString() }
   245→      setSelectedTicket(updated)
   246→      setTickets(prev => prev.map(t => t.id === updated.id ? updated : t))
   247→      toast.success(`Ticket status changed to ${statusConfig[newStatus as TicketStatus]?.label || newStatus}`)
   248→    } finally {
   249→      setChangingStatus(false)
   250→    }
   251→  }
   252→
   253→  const handleCloseTicket = async () => {
   254→    if (!selectedTicket) return
   255→    await handleStatusChange('CLOSED')
   256→  }
   257→
   258→  const statusFilters = ['All', 'OPEN', 'IN_PROGRESS', 'RESOLVED', 'CLOSED']
   259→
   260→  return (
   261→    <div className="space-y-6">
   262→      {/* Stats Row */}
   263→      <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
   264→        <StatCard icon={CircleDot} label="Open Tickets" value={openCount} sub="Needs attention" color="#3B82F6" />
   265→        <StatCard icon={Clock} label="In Progress" value={inProgressCount} sub="Being worked on" color="#F59E0B" />
   266→        <StatCard icon={CheckCircle} label="Resolved" value={resolvedCount} sub="This week" color="#00a87d" />
   267→        <StatCard icon={Clock} label="Avg Response" value="< 2h" sub="Response time" color="#8B5CF6" />
   268→      </div>
   269→
   270→      {/* Filter Bar */}
   271→      <Card className="bg-card border-border rounded-xl">
   272→        <CardContent className="p-4">
   273→          <div className="flex flex-col gap-4 sm:flex-row sm:items-center">
   274→            <div className="relative flex-1">
   275→              <Search className="absolute left-3 top-1/2 size-4 -translate-y-1/2 text-muted-foreground" />
   276→              <Input
   277→                placeholder="Search tickets by ID, subject, or user..."
   278→                value={search}
   279→                onChange={(e) => { setSearch(e.target.value); setPage(1) }}
   280→                className="rounded-lg border-border bg-muted text-foreground pl-10 h-10"
   281→              />
   282→            </div>
   283→            <div className="inline-flex items-center gap-1 rounded-full bg-muted p-1 border border-border overflow-x-auto">
   284→              {statusFilters.map((f) => (
   285→                <button
   286→                  key={f}
   287→                  onClick={() => { setStatusFilter(f); setPage(1) }}
   288→                  className={`whitespace-nowrap rounded-full px-3 py-1.5 text-xs font-semibold transition-colors ${
   289→                    statusFilter === f ? 'bg-[#00D09C] text-white' : 'text-muted-foreground hover:text-foreground'
   290→                  }`}
   291→                >
   292→                  {f === 'IN_PROGRESS' ? 'In Progress' : f.charAt(0) + f.slice(1).toLowerCase()}
   293→                </button>
   294→              ))}
   295→            </div>
   296→          </div>
   297→        </CardContent>
   298→      </Card>
   299→
   300→      {/* Tickets Table */}
   301→      <Card className="bg-card border-border rounded-xl">
   302→        <CardHeader className="pb-2">
   303→          <CardTitle className="text-sm font-semibold text-foreground">
   304→            Support Tickets ({filteredTickets.length})
   305→          </CardTitle>
   306→        </CardHeader>
   307→        <CardContent>
   308→          {loading ? (
   309→            <LoadingSkeleton rows={8} />
   310→          ) : paginatedTickets.length === 0 ? (
   311→            <EmptyState icon={Ticket} title="No tickets found" description="No support tickets match the current filters." />
   312→          ) : (
   313→            <>
   314→              <div className="overflow-x-auto">
   315→                <Table>
   316→                  <TableHeader>
   317→                    <TableRow className="border-border hover:bg-transparent">
   318→                      <TableHead className="text-muted-foreground text-xs">ID</TableHead>
   319→                      <TableHead className="text-muted-foreground text-xs">User</TableHead>
   320→                      <TableHead className="text-muted-foreground text-xs">Subject</TableHead>
   321→                      <TableHead className="text-muted-foreground text-xs">Priority</TableHead>
   322→                      <TableHead className="text-muted-foreground text-xs">Status</TableHead>
   323→                      <TableHead className="text-muted-foreground text-xs hidden md:table-cell">Created</TableHead>
   324→                      <TableHead className="text-right text-muted-foreground text-xs">Actions</TableHead>
   325→                    </TableRow>
   326→                  </TableHeader>
   327→                  <TableBody>
   328→                    {paginatedTickets.map((ticket) => {
   329→                      const pCfg = priorityConfig[ticket.priority]
   330→                      const sCfg = statusConfig[ticket.status]
   331→                      const StatusIcon = sCfg.icon
   332→                      return (
   333→                        <TableRow key={ticket.id} className="border-border hover:bg-muted">
   334→                          <TableCell className="font-mono text-xs text-muted-foreground">{ticket.id}</TableCell>
   335→                          <TableCell>
   336→                            <div className="flex items-center gap-2">
   337→                              <Avatar className="size-7">
   338→                                <AvatarFallback className="bg-[#00D09C]/10 text-[#00D09C] text-[10px] font-semibold">
   339→                                  {ticket.userName.split(' ').map(n => n[0]).join('')}
   340→                                </AvatarFallback>
   341→                              </Avatar>
   342→                              <div>
   343→                                <p className="text-xs font-medium text-foreground">{ticket.userName}</p>
   344→                                <p className="text-[10px] text-muted-foreground hidden sm:block">{ticket.userEmail}</p>
   345→                              </div>
   346→                            </div>
   347→                          </TableCell>
   348→                          <TableCell className="max-w-[200px]">
   349→                            <p className="text-xs text-foreground truncate">{ticket.subject}</p>
   350→                          </TableCell>
   351→                          <TableCell>
   352→                            <Badge variant="outline" className="text-[10px] font-semibold" style={{ borderColor: `${pCfg.color}40`, backgroundColor: `${pCfg.color}12`, color: pCfg.color }}>
   353→                              {pCfg.label}
   354→                            </Badge>
   355→                          </TableCell>
   356→                          <TableCell>
   357→                            <Badge variant="outline" className="text-[10px] font-semibold gap-1" style={{ borderColor: `${sCfg.color}40`, backgroundColor: `${sCfg.color}12`, color: sCfg.color }}>
   358→                              <StatusIcon className="size-3" />
   359→                              {sCfg.label}
   360→                            </Badge>
   361→                          </TableCell>
   362→                          <TableCell className="text-[11px] text-muted-foreground hidden md:table-cell">{formatDate(ticket.createdAt)}</TableCell>
   363→                          <TableCell className="text-right">
   364→                            <Button
   365→                              variant="ghost"
   366→                              size="sm"
   367→                              className="size-7 text-muted-foreground hover:text-[#00D09C]"
   368→                              onClick={() => setSelectedTicket(ticket)}
   369→                            >
   370→                              <Eye className="size-3.5" />
   371→                            </Button>
   372→                          </TableCell>
   373→                        </TableRow>
   374→                      )
   375→                    })}
   376→                  </TableBody>
   377→                </Table>
   378→              </div>
   379→              <SimplePagination page={page} totalPages={totalPages} onPageChange={setPage} />
   380→            </>
   381→          )}
   382→        </CardContent>
   383→      </Card>
   384→
   385→      {/* Ticket Detail Dialog */}
   386→      <Dialog open={!!selectedTicket} onOpenChange={() => setSelectedTicket(null)}>
   387→        <DialogContent className="max-w-2xl max-h-[85vh] overflow-y-auto">
   388→          {selectedTicket && (
   389→            <>
   390→              <DialogHeader>
   391→                <div className="flex items-start justify-between gap-4">
   392→                  <div className="flex-1 min-w-0">
   393→                    <div className="flex items-center gap-2 mb-1.5">
   394→                      <Badge variant="outline" className="text-[10px] font-mono" style={{ borderColor: `${priorityConfig[selectedTicket.priority].color}40`, backgroundColor: `${priorityConfig[selectedTicket.priority].color}12`, color: priorityConfig[selectedTicket.priority].color }}>

   510→              </div>
   511→            </>

   515→    </div>

   517→}
