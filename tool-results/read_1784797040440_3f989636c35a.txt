   173→
   174→export function HelpSupportPage() {
   175→  const { token, user } = useAuthStore()
   176→  const { setCurrentPage } = useAppStore()
   177→
   178→  // Tab state
   179→  const [activeTab, setActiveTab] = useState<'faq' | 'contact' | 'ticket' | 'myTickets'>('faq')
   180→
   181→  // FAQ state
   182→  const [faqCategory, setFaqCategory] = useState('all')
   183→  const [expandedFaq, setExpandedFaq] = useState<number | null>(null)
   184→  const [faqSearch, setFaqSearch] = useState('')
   185→
   186→  // Ticket form state
   187→  const [ticketSubject, setTicketSubject] = useState('')
   188→  const [ticketMessage, setTicketMessage] = useState('')
   189→  const [ticketCategory, setTicketCategory] = useState('GENERAL')
   190→  const [ticketPriority, setTicketPriority] = useState('MEDIUM')
   191→  const [ticketSubmitting, setTicketSubmitting] = useState(false)
   192→
   193→  // My tickets state
   194→  const [tickets, setTickets] = useState<TicketInfo[]>([])
   195→  const [ticketsLoading, setTicketsLoading] = useState(false)
   196→  const [expandedTicket, setExpandedTicket] = useState<string | null>(null)
   197→
   198→  // Fetch tickets
   199→  const fetchTickets = useCallback(async () => {
   200→    if (!token) return
   201→    setTicketsLoading(true)
   202→    try {
   203→      const res = await fetch('/api/support/tickets', {
   204→        headers: { Authorization: `Bearer ${token}` },
   205→      })
   206→      if (res.ok) {
   207→        const data = await res.json()
   208→        setTickets(data.tickets || [])
   209→      }
   210→    } catch {
   211→      // Silent
   212→    } finally {
   213→      setTicketsLoading(false)
   214→    }
   215→  }, [token])
   216→
   217→  useEffect(() => {
   218→    if (activeTab === 'myTickets') {
   219→      fetchTickets()
   220→    }
   221→  }, [activeTab, fetchTickets])
   222→
   223→  // Submit ticket
   224→  const handleSubmitTicket = async () => {
   225→    if (!ticketSubject.trim() || !ticketMessage.trim()) {
   226→      toast.error('Please fill in all fields')
   227→      return
   228→    }
   229→    setTicketSubmitting(true)
   230→    try {
   231→      const res = await fetch('/api/support/tickets', {
   232→        method: 'POST',
   233→        headers: {
   234→          'Content-Type': 'application/json',
   235→          Authorization: `Bearer ${token}`,
   236→        },
   237→        body: JSON.stringify({
   238→          subject: ticketSubject,
   239→          message: ticketMessage,
   240→          category: ticketCategory,
   241→          priority: ticketPriority,
   242→        }),
   243→      })
   244→      if (res.ok) {
   245→        toast.success('Ticket submitted successfully! We will get back to you soon.')
   246→        setTicketSubject('')
   247→        setTicketMessage('')
   248→        setTicketCategory('GENERAL')
   249→        setTicketPriority('MEDIUM')
   250→        setActiveTab('myTickets')
   251→      } else {
   252→        const data = await res.json()
   253→        toast.error(data.error || 'Failed to submit ticket')
   254→      }
   255→    } catch {
   256→      toast.error('Failed to submit ticket')
   257→    } finally {
   258→      setTicketSubmitting(false)
   259→    }
   260→  }
   261→
   262→  // Close ticket
   263→  const handleCloseTicket = async (ticketId: string) => {
   264→    try {
   265→      const res = await fetch(`/api/support/tickets/${ticketId}`, {
   266→        method: 'DELETE',
   267→        headers: { Authorization: `Bearer ${token}` },
   268→      })
   269→      if (res.ok) {
   270→        toast.success('Ticket closed successfully')
   271→        fetchTickets()
   272→      }
   273→    } catch {
   274→      toast.error('Failed to close ticket')
   275→    }
   276→  }
   277→
   278→  // Filtered FAQs
   279→  const filteredFaqs = FAQ_DATA.filter(faq => {
   280→    const matchesCategory = faqCategory === 'all' || faq.category === faqCategory
   281→    const matchesSearch = !faqSearch || faq.question.toLowerCase().includes(faqSearch.toLowerCase()) || faq.answer.toLowerCase().includes(faqSearch.toLowerCase())
   282→    return matchesCategory && matchesSearch
   283→  })
   284→
   285→  let si = 0
   286→
   287→  return (
   288→    <div className="p-4 sm:p-6 lg:p-8 space-y-6 max-w-3xl mx-auto bg-background min-h-screen">
   289→      {/* ── Header ──────────────────────────────────────────── */}
   290→      <motion.div custom={si++} variants={fadeInUp} initial="hidden" animate="visible">
   291→        <div className="flex items-center gap-3 mb-1">
   292→          <button
   293→            onClick={() => setCurrentPage('profile')}
   294→            className="size-9 rounded-lg bg-card border border-border flex items-center justify-center hover:bg-muted transition-colors"
   295→          >
   296→            <ArrowLeft className="size-4 text-muted-foreground" />
   297→          </button>
   298→          <div className="flex items-center gap-2.5">
   299→            <div className="size-9 rounded-lg bg-[#5367ff]/10 flex items-center justify-center">
   300→              <HelpCircle className="size-5 text-[#5367ff]" />
   301→            </div>
   302→            <div>
   303→              <h1 className="text-lg font-bold text-foreground">Help & Support</h1>
   304→              <p className="text-xs text-muted-foreground">We&apos;re here to help you</p>
   305→            </div>
   306→          </div>
   307→        </div>
   308→      </motion.div>
   309→
   310→      {/* ── Quick Actions ────────────────────────────────────── */}
   311→      <motion.div custom={si++} variants={fadeInUp} initial="hidden" animate="visible">
   312→        <div className="grid grid-cols-3 gap-3">
   313→          <button
   314→            onClick={() => setActiveTab('faq')}
   315→            className={cn(
   316→              "flex flex-col items-center gap-2 p-4 rounded-2xl border-2 transition-all",
   317→              activeTab === 'faq'
   318→                ? "border-[#5367ff] bg-[#5367ff]/5"
   319→                : "border-border bg-card hover:border-[#5367ff]/30 hover:bg-[#5367ff]/5"
   320→            )}
   321→          >
   322→            <div className={cn(
   323→              "size-10 rounded-xl flex items-center justify-center",
   324→              activeTab === 'faq' ? "bg-[#5367ff]" : "bg-[#5367ff]/10"
   325→            )}>
   326→              <HelpCircle className={cn("size-5", activeTab === 'faq' ? "text-white" : "text-[#5367ff]")} />
   327→            </div>
   328→            <div className="text-center">
   329→              <p className={cn("text-xs font-semibold", activeTab === 'faq' ? "text-[#5367ff]" : "text-foreground")}>FAQ</p>
   330→              <p className="text-[10px] text-muted-foreground mt-0.5">Quick answers</p>
   331→            </div>
   332→          </button>
   333→
   334→          <button
   335→            onClick={() => setActiveTab('contact')}
   336→            className={cn(
   337→              "flex flex-col items-center gap-2 p-4 rounded-2xl border-2 transition-all",
   338→              activeTab === 'contact'
   339→                ? "border-[#00D09C] bg-[#00D09C]/5"

   391→              />
   392→              {faqSearch && (
