   183→function formatDate(dateStr: string, range: RangeOption): string {
   184→  const d = new Date(dateStr)
   185→  if (range === '1D') return d.toLocaleTimeString('en-IN', { hour: '2-digit', minute: '2-digit', hour12: false })
   186→  if (range === '1W') return d.toLocaleDateString('en-IN', { weekday: 'short', hour: '2-digit', minute: '2-digit', hour12: false })
   187→  if (range === '1M' || range === '3M') return d.toLocaleDateString('en-IN', { month: 'short', day: 'numeric' })
   188→  return d.toLocaleDateString('en-IN', { month: 'short', year: '2-digit' })
   189→}
   190→
   191→function formatExpiry(dateStr: string): string {
   192→  if (!dateStr) return '--'
   193→  const d = new Date(dateStr)
   194→  return d.toLocaleDateString('en-IN', { day: '2-digit', month: 'short', year: 'numeric' })
   195→}

   212→function CustomTooltip({ active, payload, range }: { active?: boolean; payload?: Array<{ payload: CandleData }>; range: RangeOption }) {
   213→  if (!active || !payload || !payload.length) return null
   214→  const d = payload[0].payload
   215→  const isUp = d.close >= d.open
   216→
   217→  return (
   218→    <div className="bg-card border border-border rounded-lg p-3 shadow-xl text-xs">
   219→      <div className="font-semibold text-foreground mb-1.5">{formatDate(d.date, range)}</div>
   220→      <div className="grid grid-cols-2 gap-x-4 gap-y-0.5">
   221→        <span className="text-muted-foreground">Open</span>
   222→        <span className="font-mono font-tabular text-right">{formatPrice(d.open)}</span>
   223→        <span className="text-muted-foreground">High</span>
   224→        <span className="font-mono font-tabular text-right">{formatPrice(d.high)}</span>
   225→        <span className="text-muted-foreground">Low</span>
   226→        <span className="font-mono font-tabular text-right">{formatPrice(d.low)}</span>
   227→        <span className="text-muted-foreground">Close</span>
   228→        <span className={`font-mono font-tabular text-right font-semibold ${isUp ? 'text-[#00B386]' : 'text-[#EB5B3C]'}`}>
   229→          {formatPrice(d.close)}
   230→        </span>
   231→        {d.volume > 0 && (
   232→          <>
   233→            <span className="text-muted-foreground">Volume</span>
   234→            <span className="font-mono font-tabular text-right">{formatVolume(d.volume)}</span>
   235→          </>
   236→        )}
   237→      </div>
   238→    </div>
   239→  )
   240→}

   244→function RangeBar({ label, low, high, current, lowLabel, highLabel }: {
   245→  label: string
   246→  low: number
   247→  high: number
   248→  current: number
   249→  lowLabel?: string
   250→  highLabel?: string
   251→}) {
   252→  const range = high - low
   253→  const position = range > 0 ? Math.min(100, Math.max(0, ((current - low) / range) * 100)) : 50
   254→
   255→  return (
   256→    <div className="space-y-2">
   257→      <div className="flex items-center justify-between">
   258→        <span className="text-xs text-muted-foreground">{lowLabel || label + ' Low'}</span>
   259→        <span className="text-xs text-muted-foreground">{highLabel || label + ' High'}</span>
   260→      </div>
   261→      <div className="flex items-center justify-between text-sm font-mono font-tabular font-semibold">
   262→        <span className="text-[#EB5B3C]">{formatPrice(low)}</span>
   263→        <span className="text-[#00B386]">{formatPrice(high)}</span>
   264→      </div>
   265→      <div className="h-2 rounded-full bg-muted relative overflow-hidden">
   266→        <div
   267→          className="absolute top-0 left-0 h-full rounded-full"
   268→          style={{
   269→            width: '100%',
   270→            background: 'linear-gradient(to right, #eb5b3c, #00d09c)',
   271→            opacity: 0.25,
   272→          }}
   273→        />
   274→        <div
   275→          className="absolute top-1/2 -translate-y-1/2 size-3 bg-[#00d09c] border-2 border-white rounded-full shadow-sm z-10"
   276→          style={{ left: `calc(${position}% - 6px)` }}
   277→        />
   278→      </div>
   279→    </div>
   280→  )
   281→}

   315→          <ChevronDown className="size-4 text-muted-foreground" />

   320→          <motion.div
   321→            initial={{ height: 0, opacity: 0 }}

   325→            className="overflow-hidden"
   326→          >
   327→            <div className="px-5 pb-4">
   328→              {children}
   329→            </div>
   330→          </motion.div>
   331→        )}
   332→      </AnimatePresence>
   333→    </div>
   334→  )
