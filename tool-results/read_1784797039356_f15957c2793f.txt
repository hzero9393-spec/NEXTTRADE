   114→function formatDate(dateStr: string | null): string {
   115→  if (!dateStr) return '—'
   116→  return new Date(dateStr).toLocaleDateString('en-IN', {
   117→    day: '2-digit',
   118→    month: 'long',
   119→    year: 'numeric',
   120→  })
   121→}
   122→
   123→function getInitials(name: string | null | undefined): string {
   124→  if (!name) return '??'
   125→  return name
   126→    .split(' ')
   127→    .map(n => n[0])
   128→    .join('')
   129→    .toUpperCase()
   130→    .slice(0, 2)
   131→}

   133→function formatRelativeTime(dateStr: string): string {
   134→  const now = new Date()
   135→  const date = new Date(dateStr)
   136→  const diffMs = now.getTime() - date.getTime()
   137→  const diffSec = Math.floor(diffMs / 1000)
   138→  const diffMin = Math.floor(diffSec / 60)
   139→  const diffHr = Math.floor(diffMin / 60)
   140→  const diffDay = Math.floor(diffHr / 24)
   141→
   142→  if (diffSec < 60) return 'Just now'
   143→  if (diffMin < 60) return `${diffMin} min${diffMin !== 1 ? 's' : ''} ago`
   144→  if (diffHr < 24) return `${diffHr} hour${diffHr !== 1 ? 's' : ''} ago`
   145→  if (diffDay < 7) return `${diffDay} day${diffDay !== 1 ? 's' : ''} ago`
   146→  return date.toLocaleDateString('en-IN', { day: '2-digit', month: 'short', year: 'numeric' })
   147→}

   178→// ─── Image Compression Helper ────────────────────────────────────

   180→function compressImage(file: File, maxWidth: number, maxHeight: number, quality: number): Promise<string> {
   181→  return new Promise((resolve, reject) => {
   182→    const reader = new FileReader()
   183→    reader.onload = (e) => {
   184→      const img = new Image()
   185→      img.onload = () => {
   186→        const canvas = document.createElement('canvas')
   187→        let width = img.width
   188→        let height = img.height
   189→
   190→        // Resize if larger than max dimensions
   191→        if (width > maxWidth || height > maxHeight) {
   192→          const ratio = Math.min(maxWidth / width, maxHeight / height)
   193→          width = Math.round(width * ratio)
   194→          height = Math.round(height * ratio)
   195→        }
   196→
   197→        canvas.width = width
   198→        canvas.height = height
   199→
   200→        const ctx = canvas.getContext('2d')
   201→        if (!ctx) { reject(new Error('Canvas context failed')); return }
   202→        ctx.drawImage(img, 0, 0, width, height)
   203→
   204→        const base64 = canvas.toDataURL('image/jpeg', quality)
   205→        resolve(base64)
   206→      }
   207→      img.onerror = () => reject(new Error('Failed to load image'))
   208→      img.src = e.target?.result as string
   209→    }
   210→    reader.onerror = () => reject(new Error('Failed to read file'))
   211→    reader.readAsDataURL(file)
   212→  })
   213→}
